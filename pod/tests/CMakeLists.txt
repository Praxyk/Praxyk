SET(LIBRARY_PATH ${CMAKE_BINARY_DIR}/lib)
SET(PYTHON_PATH ${CMAKE_BINARY_DIR}/python)

MACRO(PRAXYK_ADD_PYTHON_TEST test)
    SET(TEST_FILE "${CMAKE_CURRENT_SOURCE_DIR}/${test}.py")
    CONFIGURE_FILE(
        ${CMAKE_CURRENT_SOURCE_DIR}/unit_test_template.sh.in
        ${CMAKE_CURRENT_BINARY_DIR}/${test}.sh
    @ONLY)
    ADD_TEST(${test} ${CMAKE_CURRENT_BINARY_DIR}/${test}.sh)
ENDMACRO(PRAXYK_ADD_PYTHON_TEST)

SET(PRAXYK_PYTHON_TESTS
    pod_facial_rec_test
    pod_ocr_test
    pod_spam_test
)

FOREACH(test ${PRAXYK_PYTHON_TESTS})
    PRAXYK_ADD_PYTHON_TEST(${test})
ENDFOREACH(test ${PRAXYK_PYTHON_TESTS})
